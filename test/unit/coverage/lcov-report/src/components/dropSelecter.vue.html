<template>
  <div>
    <p @click="open">{{ selected }}</p>
    <ul :class="{ 'open':isOpen }">
      <li v-for="option in options" v-if="selected!==option" @click="select(option)">{{ option }}</li>
    </ul>
  </div>
</template>
 
<script>
  export default {
    name: 'dropSelecter',
    props: [
      'options',
      'default'
    ],
    mounted () {
      this.selected = this.default ? this.options[this.default] : this.options[0]
    },
    data () {
      return {
        selected: '',
        isOpen: false
      }
    },
    methods: {
      open () {
        this.isOpen = !this.isOpen
      },
      select (option) {
        this.isOpen = false
        this.selected = option
        this.$emit('input', this.selected)
      }
    }
  }
</script>
